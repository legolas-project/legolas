8 !> @note
9 !> We don't use rj_low (= r_{j-1}), rj_center (= r_j) or rj_high (= r_{j+1}),
10 !! because in every grid block [i, i+1] there is a contribution from both
11 !! the current functions in this interval as from the functions in the grid
12 !! block [i-1, i]. See page 193 of advanded MHD by Keppens and Poedts, this
13 !! holds both for cubic and quadratic functions.
14 !! Hence we use rj_lo for r_j and rj_hi for r_{j+1}, which in the interval
15 !! [i, i+1] (2 and 4) corresponds to the current quadratic/cubic functions,
16 !! while in the interval [i-1, i] (1 and 3) this corresponds to the previous
17 !! quadratic/cubic functions.
18 !
19 ! DESCRIPTION:
20 !> Module to calculate the quadratic and cubic spline functions and their
21 !! derivatives.
22 !
23 module mod_spline_functions

31  !> Calculates the quadratic basis functions.
32  !! @param[in] r The current value for r in grid_gauss
33  !! @param[in] rj_lo Left value of r in grid
34  !! @param[in] rj_hi Right value of r in grid (so r has 4 values between
35  !! rj_hi and rj_lo)
36  !! @param[out] h_quadratic Array containing the 4 quadratic basis functions
37  subroutine quadratic_factors(r, rj_lo, rj_hi, h_quadratic)
38  real(dp), intent(in) :: r, rj_lo, rj_hi
39  real(dp), intent(out) :: h_quadratic(4)
40 
41  h_quadratic(1) = 4.0d0 * (r - rj_lo) * (rj_hi - r) / (rj_hi - rj_lo)**2
42  h_quadratic(2) = 0.0d0
43  h_quadratic(3) = (2.0d0*r - rj_hi - rj_lo) * (r - rj_lo) / (rj_hi - rj_lo)**2
44  h_quadratic(4) = (2.0d0*r - rj_hi - rj_lo) * (r - rj_hi) / (rj_hi - rj_lo)**2
45 
46  end subroutine quadratic_factors

48  !> Calculates the derivatives of the quadratic basis functions.
49  !! @param[in] r The current value for r in grid_gauss
50  !! @param[in] rj_lo Left value of r in grid
51  !! @param[in] rj_hi Right value of r in grid (so r has 4 values between
52  !! rj_hi and rj_lo)
53  !! @param[out] dh_quadratic_dr Array containing the derivatives of the
54  !! 4 quadratic basis functions
55  subroutine quadratic_factors_deriv(r, rj_lo, rj_hi, dh_quadratic_dr)
56  real(dp), intent(in) :: r, rj_lo, rj_hi
57  real(dp), intent(out) :: dh_quadratic_dr(4)
58 
59  dh_quadratic_dr(1) = 4.0d0 * (-2.0d0*r + rj_hi + rj_lo) / (rj_hi - rj_lo)**2
60  dh_quadratic_dr(2) = 0.0d0
61  dh_quadratic_dr(3) = (4.0d0*r - rj_hi - 3.0d0*rj_lo) / (rj_hi - rj_lo)**2
62  dh_quadratic_dr(4) = (4.0d0*r - rj_lo - 3.0d0*rj_hi) / (rj_hi - rj_lo)**2
63 
64  end subroutine quadratic_factors_deriv

66  !> Calculates the cubic basis functions.
67  !! @param[in] r The current value for r in grid_gauss
68  !! @param[in] rj_lo Left value of r in grid
69  !! @param[in] rj_hi Right value of r in grid (so r has 4 values between
70  !! rj_hi and rj_lo)
71  !! @param[out] h_cubic Array containing the 4 cubic basis functions
72  subroutine cubic_factors(r, rj_lo, rj_hi, h_cubic)
73  real(dp), intent(in) :: r, rj_lo, rj_hi
74  real(dp), intent(out) :: h_cubic(4)
75 
76  h_cubic(1) = 3.0d0 * ( (r - rj_lo) / (rj_hi - rj_lo) )**2 &
77  -2.0d0 * ( (r - rj_lo) / (rj_hi - rj_lo) )**3
78  h_cubic(2) = 3.0d0 * ( (rj_hi - r) / (rj_hi - rj_lo) )**2 &
79  -2.0d0 * ( (rj_hi - r) / (rj_hi - rj_lo) )**3
80  h_cubic(3) = (r - rj_hi) * ( (r - rj_lo) / (rj_hi - rj_lo) )**2
81  h_cubic(4) = (r - rj_lo) * ( (rj_hi - r) / (rj_hi - rj_lo) )**2
82 
83  end subroutine cubic_factors

94  real(dp), intent(out) :: dh_cubic_dr(4)
95 
96  dh_cubic_dr(1) = 6.0d0 * (r - rj_lo) / (rj_hi - rj_lo)**2 &
97  -6.0d0 * (r - rj_lo)**2 / (rj_hi - rj_lo)**3
98  dh_cubic_dr(2) = -6.0d0 * (rj_hi - r) / (rj_hi - rj_lo)**2 &
99  +6.0d0 * (rj_hi - r)**2 / (rj_hi - rj_lo)**3
100  dh_cubic_dr(3) = ( 2.0d0*(r - rj_hi) * (r - rj_lo) + (r - rj_lo)**2 ) &
101  / (rj_hi - rj_lo)**2
102  dh_cubic_dr(4) = ( 2.0d0*(r - rj_lo) * (r - rj_hi) + (r - rj_hi)**2 ) &
103  / (rj_hi - rj_lo)**2
104 
105  end subroutine cubic_factors_deriv
