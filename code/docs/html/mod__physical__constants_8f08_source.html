137  !> Calculates the other unit normalisations based on the pre-defined
138  !! unit_length, unit_numberdensity and unit_temperature OR unit_velocity.
139  !! @param[in] cgs_units Whether to use cgs units or not.
140  subroutine set_normalisations(cgs_units)
141  logical, intent(in) :: cgs_units
142  real(real64) :: mp, kB, mu0
143 
144  if (cgs_units) then
145  mp = mp_cgs
146  kb = kb_cgs
147  mu0 = 4.0d0*dpi
148  else
149  mp = mp_si
150  kb = kb_si
151  mu0 = mu0_si
152  end if
153 
154  unit_density = (1.0d0 + 4.0d0*he_abundance) * mp * unit_numberdensity
155 
156  ! if unit_velocity = 0 (L, nb and T defined) -- floating point comparison
157  if (abs(unit_velocity - 0.0d0) < 1.0d-9) then
158  unit_pressure = (2.0d0 + 3.0d0*he_abundance) * &
159  unit_numberdensity * kb * unit_temperature
160  unit_velocity = sqrt(unit_pressure / unit_density)
161  ! unit velocity != 0 (L, nb and v defined)
162  else
163  unit_pressure = unit_density * unit_velocity**2
164  unit_temperature = unit_pressure / ((2.0d0 + 3.0d0*he_abundance) * &
165  unit_numberdensity * kb)
166  end if
167 
168  unit_magneticfield = sqrt(mu0 * unit_pressure)
169  unit_time = unit_length / unit_velocity
170 
171  unit_luminosity = unit_pressure / ((1.0d0 + 2.0d0*he_abundance) * &
172  (unit_numberdensity**2 * unit_time))
173 
174  unit_dldt = unit_luminosity / unit_temperature
175 
176  unit_conduction = unit_density * unit_length * unit_velocity**3 / &
177  unit_temperature
178 
179  unit_dtc_drho = unit_conduction / unit_density
180  unit_dtc_dt = unit_conduction / unit_temperature
181  unit_dtc_db2 = unit_conduction / (unit_magneticfield**2)
182 
183  end subroutine set_normalisations
