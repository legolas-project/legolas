53  quadblock_idx = 0

55  ! Iterate over gridpoints to calculate blocks
56  do i = 1, gridpts-1
57  ! Set quadblock to zero (quadblock is complex, so tuple)

59

61  r_lo = grid(i)
62  r_hi = grid(i + 1)
63 
64  do j = 1, n_gauss
65  ! Current grid index (from 4*n_gauss points)
66  gauss_idx = (i - 1)*n_gauss + j
67  r = grid_gauss(gauss_idx)
68  if ((geometry .eq. "cartesian") .or. (geometry .eq. "Cartesian")) then
69  eps = 1.0d0
70  else if (geometry .eq. "cylindrical") then
71  eps = r
72  else
73  write(*, *) "Geometry not defined correctly."
74  write(*, *) "Currently set on: ", geometry
75  stop
76  endif
77 
78  curr_weight = gaussian_weights(j)
79 
80  call quadratic_factors(r, r_lo, r_hi, h_quadratic)
81  call cubic_factors(r, r_lo, r_hi, h_cubic)
82 
83  call get_b_elements(gauss_idx, eps, curr_weight, quadblock)
84 
85  end do ! end do iteration gaussian points
86 
87  !! Solving in [0, 1], so scale with difference
88  quadblock = quadblock * (r_hi - r_lo)
89 
90  !! Apply boundary conditions on edges
91  if (i == 1) then
92  call boundaries_b_left_edge(quadblock)
93  end if
94  if (i == gridpts - 1) then
95  call boundaries_b_right_edge(quadblock)
96  end if
97 
98  ! Fill B-matrix with quadblock entries
99  do k = 1, dim_quadblock
100  do l = 1, dim_quadblock
101  idx1 = k + quadblock_idx
102  idx2 = l + quadblock_idx
103  matrix_b(idx1, idx2) = matrix_b(idx1, idx2) + real(quadblock(k, l))
104  end do
105  end do
106  !! This ensures that the quadblock is shifted on the main diagonal using

108  !! dim_quadblock (= 32), as the bottom-right part of the quadblock
109  !! overlaps with the top-left part when shifting along the main diagonal.
110  quadblock_idx = quadblock_idx + dim_subblock
