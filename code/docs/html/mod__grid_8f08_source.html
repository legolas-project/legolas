55  !> Sets up grid_gauss, that is, the grid evaluated in the four
56  !! Gaussian points.
57  subroutine set_grid_gauss()
58  real(dp) :: x_lo, x_hi, dx, xi(n_gauss)
59  integer :: i, j, idx
60 
61  ! Check for origin in cylindrical coordinates
62  if (geometry == "cylindrical") then
63  grid_gauss(1) = 1.0d-5
64  end if
65 
66  ! Evaluate grid_gauss in nodes of Gaussian quadrature.
67  ! An integral of f(x) in [a, b] can be approximated by
68  ! 0.5*(b-a) * SUM[i from 1 -> n] ( wi * f( 0.5*(b-a)*xi + 0.5*(b-a)) )
69  ! where wi and xi are the weights and nodes at i (so 1 to 4 here).
70  ! Hence we need the gridpoints equal to the evaluation points of f(x).
71  do i = 1, gridpts - 1
72  x_lo = grid(i)
73  x_hi = grid(i + 1)
74  dx = x_hi - x_lo
75 
76  do j = 1, n_gauss
77  xi(j) = 0.5 * dx * gaussian_nodes(j) + 0.5*(x_lo + x_hi)
78  idx = (i - 1)*n_gauss + j
79  grid_gauss(idx) = xi(j)
80  end do
81  end do
82  end subroutine set_grid_gauss

84  !> Subroutine to re-grid the mesh to a non-uniform spacing.
85  !! This is based on two Gaussian curves with known widths (from
86  !! mod_global_variables); the grid is accumulated near each Gaussian maximum,
87  !! using equidistribution based on the integral under the curve defined
88  !! by the function gaussian().
89  subroutine accumulate_mesh()
90  integer :: i, integral_gridpts
91  integer :: integral_gridpts_1, integral_gridpts_2
92  real(dp) :: dx, dx_0, xi, bgf, fact, dx_eq
93  real(dp) :: gauss_xi, gauss_xi_eq
94  real(dp) :: x_sum, x_sum_prev, x_norm
95  real(dp) :: xi_weighted
96 
97  print*,"Redefining grid with mesh accumulation"
98 
99  integral_gridpts = gridpts - 1
100 
101  bgf = 0.3d0 !background
102  fact = 1.0d0
103 
104  ! first evaluation of integral to get weighted values
105  integral_gridpts_1 = 2*integral_gridpts + 1
106  dx = (grid(gridpts) - grid(1)) / float(integral_gridpts_1 - 1)
107  xi = grid(1)
108  x_sum = 0.0d0
109 
110  do i = 1, integral_gridpts_1
111  gauss_xi = gaussian(xi, bgf, fact)
112  x_sum = x_sum + (gauss_xi * dx)
113  xi = xi + dx
114  end do
115  x_norm = (grid(gridpts) - grid(1)) / x_sum
116 
117  ! second evaluation of integral using weighted points
118  integral_gridpts_2 = 50*integral_gridpts
119  dx_eq = (grid(gridpts) - grid(1)) / float(integral_gridpts)
120  xi = grid(1)
121  x_sum = 0.0d0 ! x0 here
122  gauss_xi = gaussian(xi, bgf, fact) * x_norm ! This is at x0 for now
123  dx_0 = (grid(gridpts) - grid(1)) * gauss_xi / float(integral_gridpts_2)
124 
125 
126  do i = 2, integral_gridpts
127  gauss_xi_eq = float(i - 1) * dx_eq + grid(1)
128  do while (gauss_xi_eq > x_sum) !x_sum is 0 at first pass
129  dx = dx_0 / gauss_xi
130  xi = xi + dx
131  x_sum_prev = x_sum
132  gauss_xi = gaussian(xi, bgf, fact)
133  gauss_xi = gauss_xi * x_norm
134  x_sum = x_sum + (gauss_xi * dx)
135  end do
136 
137  xi_weighted = (gauss_xi_eq - x_sum_prev) / (x_sum - x_sum_prev)
138 
139  ! Re-define grid
140  grid(i) = xi - dx*(1.0d0 - xi_weighted)
141  end do
142 
143  ! Ensure correct end points and final spacing
144  grid(1) = x_start
145  grid(integral_gridpts+1) = x_end
146  grid(integral_gridpts) = 0.5 * (grid(integral_gridpts - 1) &
147  + grid(integral_gridpts + 1))
148 
149  end subroutine accumulate_mesh

153  !> Function to calculate a Gaussian curve based on known widths and
154  !! expected values (from mod_global_variables). The Gaussian is evaluated
155  !! in x.
156  !! @param[in] x Value for x in the Gaussian function
157  !! @param[in] bgf Value for the background field, hardcoded to 0.3
158  !! @param[in] fact Division factor in the Gaussian function, hardcoded to 1.0
159  !! @return f_gauss Value of the Gaussian function, evaluated in x
160  function gaussian(x, bgf, fact) result(f_gauss)
161  use mod_physical_constants
162 
163  real(dp), intent(in) :: x, bgf, fact
164  real(dp) :: f_gauss
165  real(dp) :: gauss_1, gauss_2, norm_1, norm_2
166 
167  norm_1 = 1.0d0 / (sigma_1 * sqrt(2*dpi))
168  norm_2 = 1.0d0 / (sigma_2 * sqrt(2*dpi))
169 
170  gauss_1 = norm_1 * exp(-0.5d0 * ((x - ev_1) / sigma_1)**2)
171  gauss_2 = norm_2 * exp(-0.5d0 * ((x - ev_2) / sigma_2)**2)
172 
173  f_gauss = bgf + (1.0d0 - bgf) * (gauss_1 + fact*gauss_2) / fact
174 
175  end function gaussian
