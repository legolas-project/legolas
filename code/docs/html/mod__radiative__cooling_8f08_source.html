174  !> Interpolates the rudimentary cooling curves using ncool points.
175  !! A second-order polynomial interpolation is used except near sharp jumps.
176  !! @param[in] ntable number of entries in the cooling table.
177  !! @param[in] table_T Temperature entries of cooling table.
178  !! @param[in] table_L Luminosity entries of cooling table.
179  subroutine interpolate_cooling_curve(ntable, table_T, table_L)
180  use mod_physical_constants
181 
182  integer, intent(in) :: ntable
183  real(dp), intent(in) :: table_T(:), table_L(:)
184  real(dp) :: fact1, fact2, fact3, dL1, dL2, ratt
185  integer :: i, j
186  logical :: jump
187 
188  max_t = table_t(ntable)
189  min_t = table_t(1)
190  ratt = (max_t-min_t) / dble(ncool-1)
191 
192  interp_table_t(1) = min_t
193  interp_table_l(1) = table_l(1)
194 
195  interp_table_t(ncool) = max_t
196  interp_table_l(ncool) = table_l(ntable)
197 
198  do i=2, ncool ! loop to create one table
199  interp_table_t(i) = interp_table_t(i-1) + ratt
200  do j=1,ntable-1 ! loop to create one spot on a table
201  ! Second order polynomial interpolation, except at the outer edge,
202  ! or in case of a large jump.
203  if (interp_table_t(i) < table_t(j+1)) then
204  if (j.eq. ntable-1 ) then
205  fact1 = (interp_table_t(i)-table_t(j+1)) &
206  /(table_t(j)-table_t(j+1))
207 
208  fact2 = (interp_table_t(i)-table_t(j)) &
209  /(table_t(j+1)-table_t(j))
210 
211  interp_table_l(i) = table_l(j)*fact1 + table_l(j+1)*fact2
212  exit
213  else
214  dl1 = table_l(j+1)-table_l(j)
215  dl2 = table_l(j+2)-table_l(j+1)
216  jump =(max(dabs(dl1),dabs(dl2)) > 2*min(dabs(dl1),dabs(dl2)))
217  endif
218 
219  if( jump ) then
220  fact1 = (interp_table_t(i)-table_t(j+1)) &
221  /(table_t(j)-table_t(j+1))
222 
223  fact2 = (interp_table_t(i)-table_t(j)) &
224  /(table_t(j+1)-table_t(j))
225 
226  interp_table_l(i) = table_l(j)*fact1 + table_l(j+1)*fact2
227  exit
228  else
229  fact1 = ((interp_table_t(i)-table_t(j+1)) &
230  * (interp_table_t(i)-table_t(j+2))) &
231  / ((table_t(j)-table_t(j+1)) &
232  * (table_t(j)-table_t(j+2)))
233 
234  fact2 = ((interp_table_t(i)-table_t(j)) &
235  * (interp_table_t(i)-table_t(j+2))) &
236  / ((table_t(j+1)-table_t(j)) &
237  * (table_t(j+1)-table_t(j+2)))
238 
239  fact3 = ((interp_table_t(i)-table_t(j)) &
240  * (interp_table_t(i)-table_t(j+1))) &
241  / ((table_t(j+2)-table_t(j)) &
242  * (table_t(j+2)-table_t(j+1)))
243 
244  interp_table_l(i) = table_l(j)*fact1 + table_l(j+1)*fact2 &
245  + table_l(j+2)*fact3
246  exit
247  endif
248  endif
249  enddo ! end loop to find create one spot on a table
250  enddo ! end loop to create one table
251 
252  ! Go from logarithmic to actual values.
253  interp_table_t(1:ncool) = 10.0d0**interp_table_t(1:ncool)
254  interp_table_l(1:ncool) = 10.0d0**interp_table_l(1:ncool)
255 
256  ! Normalise values
257  interp_table_t(1:ncool) = interp_table_t(1:ncool) / unit_temperature
258  interp_table_l(1:ncool) = interp_table_l(1:ncool) / unit_luminosity
259 
260  lgmin_t = dlog10(min_t)
261  lgmax_t = dlog10(max_t)
262  lgstep = (lgmax_t-lgmin_t) * 1.d0 / (ncool-1)
263 
264  interp_table_dldt(1) = (interp_table_l(2) - interp_table_l(1)) &
265  / (interp_table_t(2) - interp_table_t(1))
266  interp_table_dldt(ncool) = (interp_table_l(ncool) - interp_table_l(ncool-1)) &
267  / (interp_table_t(ncool) - interp_table_t(ncool-1))
268 
269  do i=2,ncool-1
270  interp_table_dldt(i) = (interp_table_l(i+1) - interp_table_l(i-1)) &
271  / (interp_table_t(i+1) - interp_table_t(i-1))
272  enddo
273 
274  end subroutine interpolate_cooling_curve
