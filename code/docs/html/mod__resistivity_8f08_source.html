30  real(dp) :: ec, me, e0, kB, eta_1MK

32  call get_constants(ec, me, e0, kb)

34  eta = dpi * z_ion * ec**2 * sqrt(me) * log(coulomb_log) / &
35  ((4 * dpi * e0)**2 * (kb * t0)**1.5d0)

37  !! Set the unit resistivity, such that the normalised resistivity
38  !! at 1 MK equals approximately 0.1. This can be done, as a unit current
